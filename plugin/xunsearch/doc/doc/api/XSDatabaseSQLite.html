public function query($sql)
{
    //echo "[DEBUG] SQL: $sql\n";
    $res = sqlite_query($this->link, $sql);
    if ($res === false) {
        throw new XSException('SQLITE ERROR: ' . sqlite_error_string($this->link));
    }
    if (!is_resource($res)) {
        $ret = $res;
    } else {
        $ret = array();
        while ($tmp = sqlite_fetch_array($res, SQLITE_ASSOC)) {
            $ret[] = $tmp;
        }
    }
    return $ret;
}
